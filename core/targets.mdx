## Overview

Targets are the final stage of a Subsquid Pipes pipeline. They handle **data persistence**, **cursor management**, and **fork recovery**. Targets receive batches of processed data and write them to databases, message queues, files, or any custom sink.

## Creating a Target

The simplest target implements a `write` function:

```typescript
import { createTarget } from '@subsquid/pipes/core'

const target = createTarget({
  write: async ({ read, logger }) => {
    for await (const batch of read()) {
      // Process batch.data
      console.log(`Received ${batch.data.transfers.length} transfers`)
      
      // Save to database, send to queue, etc.
      await saveToDatabase(batch.data)
      
      // Save cursor for resume capability
      await saveCursor(batch.ctx.state.current)
    }
  }
})

// Connect to pipeline
await source.pipe(transformer).pipeTo(target)
```

## Target Interface

**Type definition from source:**
```typescript
// target.ts:5-11
type Target<In> = {
  write: (writer: {
    read: (cursor?: BlockCursor) => AsyncIterableIterator<PortalBatch<In>>
    logger: Logger
  }) => Promise<void>
  fork?: (previousBlocks: BlockCursor[]) => Promise<BlockCursor | null>
}
```

### write (required)

The `write` function receives:
- `read`: Function to get batches (optionally from a cursor)
- `logger`: Pino logger instance

```typescript
{
  write: async ({ read, logger }) => {
    // Load last saved position
    const lastCursor = await loadLastCursor()
    
    logger.info(`Resuming from block ${lastCursor?.number || 0}`)
    
    // Read from saved cursor
    for await (const batch of read(lastCursor)) {
      await processBatch(batch)
      
      // Save progress
      await saveCursor(batch.ctx.state.current)
    }
    
    logger.info('Sync complete')
  }
}
```

### fork (optional)

Handles blockchain reorganizations:

```typescript
{
  write: async ({ read, logger }) => {
    for await (const batch of read()) {
      await save(batch)
    }
  },
  
  fork: async (previousBlocks) => {
    // previousBlocks: valid chain from Portal
    // Return cursor to resume from
    
    const safeCursor = await rollbackToSafeBlock(previousBlocks)
    return safeCursor
  }
}
```

See [Fork Handling](/core/fork-handling) for details.

## Built-in Targets

### ClickHouse Target

High-performance analytics database:

```typescript
import { createClient } from '@clickhouse/client'
import { clickhouseTarget } from '@subsquid/pipes/targets/clickhouse'

const client = createClient({
  url: 'http://localhost:8123',
  username: 'default',
  password: 'default'
})

await source
  .pipe(decoder)
  .pipeTo(
    clickhouseTarget({
      client,
      settings: {
        table: 'transfers',  // Progress tracking table
        database: 'default'
      },
      onData: async ({ data, ctx }) => {
        // Insert data
        await client.insert({
          table: 'erc20_transfers',
          values: data.transfers,
          format: 'JSONEachRow'
        })
      },
      onRollback: async ({ from, to, ctx }) => {
        // Delete forked data
        await client.query({
          query: `
            DELETE FROM erc20_transfers
            WHERE block_number > ${from.number}
            AND block_number <= ${to.number}
          `
        })
      }
    })
  )
```

**Key features:**
- Automatic cursor persistence
- Fork handling with rollback support
- Batch optimization
- Progress tracking

### Drizzle Target

Type-safe SQL with Drizzle ORM:

```typescript
import { drizzle } from 'drizzle-orm/node-postgres'
import { drizzleTarget } from '@subsquid/pipes/targets/drizzle'
import { Pool } from 'pg'
import * as schema from './schema'

const pool = new Pool({
  connectionString: process.env.DATABASE_URL
})

const db = drizzle(pool, { schema })

await source
  .pipe(decoder)
  .pipeTo(
    drizzleTarget({
      db,
      onData: async ({ data, ctx }) => {
        // Type-safe inserts
        await db.insert(schema.transfers).values(
          data.transfers.map(t => ({
            blockNumber: t.blockNumber,
            from: t.from,
            to: t.to,
            value: t.value
          }))
        )
      }
    })
  )
```

## Cursor Management

Cursors enable resuming from the last processed block:

```typescript
type BlockCursor = {
  number: number
  hash?: string
  timestamp?: number
}
```

### Saving Cursors

```typescript
const target = createTarget({
  write: async ({ read, logger }) => {
    for await (const batch of read()) {
      await processData(batch.data)
      
      // Save cursor after successful processing
      await saveCursor(batch.ctx.state.current)
    }
  }
})
```

### Loading Cursors

```typescript
const target = createTarget({
  write: async ({ read, logger }) => {
    // Load from database, file, or memory
    const cursor = await loadCursor()
    
    if (cursor) {
      logger.info(`Resuming from block ${cursor.number}`)
    }
    
    // Pass cursor to read()
    for await (const batch of read(cursor)) {
      await processData(batch.data)
      await saveCursor(batch.ctx.state.current)
    }
  }
})
```

### Example: File-based Cursor

```typescript
import { readFile, writeFile } from 'fs/promises'

async function loadCursor(): Promise<BlockCursor | undefined> {
  try {
    const data = await readFile('.cursor.json', 'utf-8')
    return JSON.parse(data)
  } catch {
    return undefined
  }
}

async function saveCursor(cursor: BlockCursor): Promise<void> {
  await writeFile('.cursor.json', JSON.stringify(cursor, null, 2))
}

const target = createTarget({
  write: async ({ read, logger }) => {
    const cursor = await loadCursor()
    
    for await (const batch of read(cursor)) {
      await processData(batch.data)
      await saveCursor(batch.ctx.state.current)
    }
  }
})
```

## Fork Handling

Targets must handle blockchain reorganizations:

```typescript
import { resolveForkCursor } from '@subsquid/pipes/core'

const target = createTarget({
  write: async ({ read, logger }) => {
    for await (const batch of read()) {
      // Save rollback chain for fork recovery
      await saveRollbackChain({
        rollbackChain: batch.ctx.state.rollbackChain,
        finalized: batch.ctx.head.finalized
      })
      
      await saveData(batch.data)
      await saveCursor(batch.ctx.state.current)
    }
  },
  
  fork: async (previousBlocks) => {
    // Load historical rollback records
    const records = loadRollbackRecords() // Newest first
    
    // Find safe cursor
    const safeCursor = await resolveForkCursor(records, previousBlocks)
    
    if (!safeCursor) {
      throw new Error('Cannot resolve fork - no common ancestor')
    }
    
    // Rollback data
    await deleteDataAfter(safeCursor.number)
    
    return safeCursor
  }
})
```

**The `resolveForkCursor` helper:**
```typescript
// fork.ts:18-21
async function resolveForkCursor(
  records: AsyncIterable<RollbackRecord> | Iterable<RollbackRecord>,
  previousBlocks: BlockCursor[]
): Promise<BlockCursor | null>
```

See [Fork Handling](/core/fork-handling) for complete details.

## Batch Processing

Each batch contains data and context:

```typescript
const target = createTarget({
  write: async ({ read, logger }) => {
    for await (const batch of read()) {
      const { data, ctx } = batch
      
      // Access processed data
      console.log('Transfers:', data.transfers.length)
      
      // Access context
      console.log('Block:', ctx.state.current.number)
      console.log('Finalized:', ctx.head.finalized?.number)
      console.log('Rollback chain:', ctx.state.rollbackChain.length)
      
      // Profiling
      const span = ctx.profiler.start('database-insert')
      await insertData(data)
      span.end()
      
      // Logging
      ctx.logger.info({ 
        blockNumber: ctx.state.current.number,
        itemCount: data.transfers.length 
      }, 'Batch processed')
    }
  }
})
```

## Error Handling

### Retry on Failure

```typescript
const target = createTarget({
  write: async ({ read, logger }) => {
    for await (const batch of read()) {
      let retries = 0
      const maxRetries = 3
      
      while (retries < maxRetries) {
        try {
          await saveData(batch.data)
          await saveCursor(batch.ctx.state.current)
          break
        } catch (error) {
          retries++
          logger.error({ error, retries }, 'Save failed, retrying...')
          
          if (retries >= maxRetries) {
            throw error
          }
          
          await sleep(1000 * retries)
        }
      }
    }
  }
})
```

### Graceful Shutdown

```typescript
let shutdownRequested = false

process.on('SIGINT', () => {
  shutdownRequested = true
})

const target = createTarget({
  write: async ({ read, logger }) => {
    for await (const batch of read()) {
      if (shutdownRequested) {
        logger.info('Shutdown requested, stopping gracefully...')
        break
      }
      
      await saveData(batch.data)
      await saveCursor(batch.ctx.state.current)
    }
  }
})
```

## Custom Targets

### Message Queue Target

```typescript
import { Kafka } from 'kafkajs'

const kafka = new Kafka({
  brokers: ['localhost:9092']
})

const producer = kafka.producer()
await producer.connect()

const kafkaTarget = createTarget({
  write: async ({ read, logger }) => {
    for await (const batch of read()) {
      // Send each transfer as a message
      await producer.send({
        topic: 'erc20-transfers',
        messages: batch.data.transfers.map(t => ({
          key: t.transactionHash,
          value: JSON.stringify(t)
        }))
      })
      
      logger.info(`Sent ${batch.data.transfers.length} messages`)
    }
    
    await producer.disconnect()
  }
})
```

### File System Target

```typescript
import { writeFile, mkdir } from 'fs/promises'
import { join } from 'path'

const fileTarget = createTarget({
  write: async ({ read, logger }) => {
    await mkdir('./output', { recursive: true })
    
    for await (const batch of read()) {
      const filename = `block-${batch.ctx.state.current.number}.json`
      const filepath = join('./output', filename)
      
      await writeFile(
        filepath,
        JSON.stringify(batch.data, null, 2)
      )
      
      logger.info(`Wrote ${filepath}`)
    }
  }
})
```

### REST API Target

```typescript
const apiTarget = createTarget({
  write: async ({ read, logger }) => {
    for await (const batch of read()) {
      // POST to external API
      const response = await fetch('https://api.example.com/transfers', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(batch.data.transfers)
      })
      
      if (!response.ok) {
        throw new Error(`API error: ${response.statusText}`)
      }
      
      logger.info(`Sent ${batch.data.transfers.length} transfers to API`)
    }
  }
})
```

## Advanced Patterns

### Batching Inserts

Buffer multiple batches before writing:

```typescript
const target = createTarget({
  write: async ({ read, logger }) => {
    let buffer: Transfer[] = []
    const BATCH_SIZE = 1000
    
    for await (const batch of read()) {
      buffer.push(...batch.data.transfers)
      
      if (buffer.length >= BATCH_SIZE) {
        await insertBatch(buffer)
        buffer = []
      }
    }
    
    // Flush remaining
    if (buffer.length > 0) {
      await insertBatch(buffer)
    }
  }
})
```

### Multiple Targets

Write to multiple destinations:

```typescript
const multiTarget = createTarget({
  write: async ({ read, logger }) => {
    for await (const batch of read()) {
      // Write to multiple places in parallel
      await Promise.all([
        saveToPostgres(batch.data),
        saveToClickHouse(batch.data),
        sendToKafka(batch.data)
      ])
      
      await saveCursor(batch.ctx.state.current)
    }
  }
})
```

### Conditional Writing

```typescript
const conditionalTarget = createTarget({
  write: async ({ read, logger }) => {
    for await (const batch of read()) {
      // Only save blocks after certain height
      if (batch.ctx.state.current.number > 18_000_000) {
        await saveData(batch.data)
      }
      
      // Always save cursor
      await saveCursor(batch.ctx.state.current)
    }
  }
})
```

## Testing Targets

Targets can be tested with mock data:

```typescript
import { describe, it, expect } from 'vitest'

const createMockBatch = (data: any, blockNumber: number) => ({
  data,
  ctx: {
    state: {
      current: { number: blockNumber, hash: '0x...' },
      rollbackChain: [],
      initial: 0,
      last: blockNumber
    },
    head: {},
    meta: {},
    query: {},
    profiler: createMockProfiler(),
    metrics: createMockMetrics(),
    logger: createMockLogger()
  }
})

describe('myTarget', () => {
  it('saves data correctly', async () => {
    const saved: any[] = []
    
    const target = createTarget({
      write: async ({ read }) => {
        for await (const batch of read()) {
          saved.push(batch.data)
        }
      }
    })
    
    const mockRead = async function* () {
      yield createMockBatch({ transfers: [1, 2, 3] }, 100)
      yield createMockBatch({ transfers: [4, 5, 6] }, 101)
    }
    
    await target.write({
      read: mockRead,
      logger: createMockLogger()
    })
    
    expect(saved).toHaveLength(2)
    expect(saved[0].transfers).toEqual([1, 2, 3])
  })
})
```

## Best Practices

### Do's

- Always save cursors after successful processing
- Implement fork handling for production use
- Use transactions for atomic writes
- Handle errors gracefully with retries
- Log progress and errors
- Test targets in isolation

```typescript
// Good: Atomic write with cursor
const target = createTarget({
  write: async ({ read, logger }) => {
    for await (const batch of read()) {
      await db.transaction(async (tx) => {
        await tx.insert(transfers).values(batch.data.transfers)
        await tx.insert(cursors).values(batch.ctx.state.current)
      })
    }
  }
})
```

### Don'ts

- Don't skip cursor persistence
- Don't ignore fork handling
- Don't block on synchronous I/O
- Don't lose data on errors

```typescript
// Bad: No cursor saved
const target = createTarget({
  write: async ({ read }) => {
    for await (const batch of read()) {
      await save(batch.data) // ❌ What if this fails?
      // No cursor saved - can't resume!
    }
  }
})

// Good: Cursor saved
const target = createTarget({
  write: async ({ read }) => {
    for await (const batch of read()) {
      await save(batch.data)
      await saveCursor(batch.ctx.state.current) // ✅
    }
  }
})
```
